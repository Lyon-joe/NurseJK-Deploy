// --- CLINICAL UTILITIES (Ported Logic) ---

/**
 * Handles exact medical dosage and infusion calculations safely.
 */
class PharmacologyCalculator {
  /**
   * Calculates fluid drop rates.
   * Drop Rate = (Total Volume * Drop Factor) / Total Time in Minutes
   */
  static calculateIvDripRate(totalVolumeMl, timeHours, dropFactorGttMl = 20) {
    if (timeHours <= 0) {
      throw new Error("Duration of infusion must be greater than zero.");
    }
    const totalTimeMinutes = timeHours * 60;
    const dropRate = (totalVolumeMl * dropFactorGttMl) / totalTimeMinutes;
    return parseFloat(dropRate.toFixed(1));
  }

  /**
   * Calculates precise liquid medication volumes based on weight.
   */
  static calculatePediatricDose(patientWeightKg, dosageMgPerKg, stockConcentrationMgPerMl) {
    if (stockConcentrationMgPerMl <= 0) {
      throw new Error("Stock concentration must be a positive non-zero value.");
    }
    const targetDoseMg = patientWeightKg * dosageMgPerKg;
    const requiredVolumeMl = targetDoseMg / stockConcentrationMgPerMl;
    
    return {
      calculatedDoseMg: parseFloat(targetDoseMg.toFixed(2)),
      administerVolumeMl: parseFloat(requiredVolumeMl.toFixed(2))
    };
  }
}

/**
 * Core rule validation for nursing decision support and triage.
 */
class ClinicalReasoningEngine {
  constructor(context) {
    this.context = context; // Expects { vitals: {...}, presentingSigns: [] }
    this.alerts = [];
    this.triageCategory = "GREEN";
  }

  evaluateVitals() {
    const v = this.context.vitals;
    if (!v) return { triageStatus: this.triageCategory, alerts: this.alerts };

    // Check for shock or severe cardiovascular compromise
    if (v.systolicBp < 90 || v.pulseRate > 120) {
      this.alerts.push("CRITICAL: Potential hemodynamic instability detected.");
      this.triageCategory = "RED";
    }
    
    // Respiratory depression or severe hypoxia
    if (v.spo2 < 92) {
      this.alerts.push("CRITICAL: Severe hypoxia. Initiate oxygen therapy protocol immediately.");
      this.triageCategory = "RED";
    }

    return { triageStatus: this.triageCategory, alerts: this.alerts };
  }

  evaluatePediatricRespiratoryStatus() {
    const v = this.context.vitals;
    const signs = this.context.presentingSigns || [];
    let assessment = { classification: "Non-Severe / Standard Care", interventions: [] };

    if (v && v.ageMonths && v.ageMonths < 60) { // Under 5 years
      const hasChestIndrawing = signs.includes("chest_indrawing");
      const hasDangerSigns = signs.some(sign => 
        ["stridor", "grunting", "inability_to_feed", "lethargy", "convulsions"].includes(sign)
      );

      // Fast breathing thresholds (WHO standard)
      let isFastBreathing = false;
      if (v.ageMonths < 12 && v.respRate >= 50) {
        isFastBreathing = true;
      } else if (v.ageMonths >= 12 && v.respRate >= 40) {
        isFastBreathing = true;
      }

      if (hasDangerSigns || v.spo2 < 90) {
        this.triageCategory = "RED";
        assessment.classification = "SEVERE PNEUMONIA / SEVERE DISEASE";
        assessment.interventions = [
          "Admit to inpatient pediatric unit immediately.",
          "Administer first dose of recommended IV antibiotics (e.g., Ampicillin + Gentamicin).",
          "Initiate closely monitored oxygen therapy."
        ];
      } else if (hasChestIndrawing || isFastBreathing) {
        assessment.classification = "PNEUMONIA";
        assessment.interventions = [
          "Prescribe appropriate course of oral Amoxicillin (e.g., 40-45 mg/kg/dose twice daily).",
          "Counsel caregiver on home care, feeding maintenance, and specific danger signs for immediate return.",
          "Schedule mandatory follow-up assessment in 48 hours."
        ];
      }
    }
    return assessment;
  }
}

const nurseSystemPrompt = `
You are a professional nurse assistant.
Always answer every clinical topic using exactly these sections and in this order:
1. Definition
2. Pathophysiology
3. Risk factors
4. Causes
5. Diagnostic assessment and findings
6. Signs and symptoms
7. Nursing management
8. Medications
9. Complications

Use safe standard clinical guidelines.
Keep the response structured under the required headings only.
`.trim();

app.get("/", (req, res) => {
  res.json({ message: "Nursing AI Assistant API is running" });
});

// New Utility Endpoint for Quick Dosage Calculations
app.post("/api/calculate/dose", (req, res) => {
  try {
    const { weight, dosagePerKg, concentration } = req.body;
    if (!weight || !dosagePerKg || !concentration) {
      return res.status(400).json({ error: "weight, dosagePerKg, and concentration fields are all required." });
    }
    const result = PharmacologyCalculator.calculatePediatricDose(weight, dosagePerKg, concentration);
    res.json(result);
  } catch (error) {
    res.status(400).json({ error: error.message });
  }
});

// New Utility Endpoint for IV Drip Rates
app.post("/api/calculate/drip-rate", (req, res) => {
  try {
    const { volume, hours, dropFactor } = req.body;
    if (!volume || !hours) {
      return res.status(400).json({ error: "volume and hours fields are required." });
    }
    const rate = PharmacologyCalculator.calculateIvDripRate(volume, hours, dropFactor || 20);
    res.json({ dropRateGttMin: rate });
  } catch (error) {
    res.status(400).json({ error: error.message });
  }
});

// Enhanced Chat Endpoint with Structural Reasoning Injection
app.post("/api/chat", async (req, res) => {
  try {
    const { message, clinicalContext } = req.body;

    if (!message || typeof message !== "string") {
      return res.status(400).json({ error: "The message field is required." });
    }

    let dynamicSystemPrompt = nurseSystemPrompt;
    let localAssessmentResults = null;

    // If clinical context data is sent along with the chat, process local clinical rules
    if (clinicalContext) {
      const engine = new ClinicalReasoningEngine(clinicalContext);
      const vitalsAnalysis = engine.evaluateVitals();
      const respiratoryAnalysis = engine.evaluatePediatricRespiratoryStatus();

      localAssessmentResults = {
        triageStatus: engine.triageCategory,
        vitalsAlerts: vitalsAnalysis.alerts,
        respiratoryClassification: respiratoryAnalysis.classification,
        suggestedInterventions: respiratoryAnalysis.interventions
      };

      // Append programmatic findings directly to the system instructions to keep the LLM completely grounded
      dynamicSystemPrompt += `\n\n[CRITICAL LOCAL ENGINE ASSESSMENT]:
The programmatic triage engine has analyzed the patient. You MUST incorporate these findings smoothly into your structured text where applicable:
- Triage Status: ${localAssessmentResults.triageStatus}
- Local Alerts: ${localAssessmentResults.vitalsAlerts.join(", ") || "None"}
- Rule-based Classification: ${localAssessmentResults.respiratoryClassification}
- Mandatory Interventions: ${localAssessmentResults.suggestedInterventions.join("; ")}
`.trim();
    }

    const response = await client.chat.completions.create({
      model: "gpt-4o-mini",
      messages: [
        { role: "system", content: dynamicSystemPrompt },
        { role: "user", content: message }
      ]
    });

    const reply = response.choices[0].message.content;
    
    // Return both the AI answer and the raw deterministic local calculations
    res.json({ 
      reply,
      engineAnalysis: localAssessmentResults
    });

  } catch (error) {
    res.status(500).json({
      error: error.message || "Server error"
    });
  }
});
